018package org.fcrepo.audit;
019
020import static org.slf4j.LoggerFactory.getLogger;
021
022import javax.annotation.PostConstruct;
023import javax.inject.Inject;
024
025import org.fcrepo.kernel.api.observer.FedoraEvent;
026
027import org.slf4j.Logger;
028
029import com.google.common.eventbus.EventBus;
030import com.google.common.eventbus.Subscribe;
031
032/**
033 * A proof-of-concept Auditor implementation that uses Logback.
034 *
035 * @author Edwin Shin
036 * @since 2014
037 */
038@Deprecated
039public class LogbackAuditor implements Auditor {
040
041    /**
042     * Logger for this class.
043     */
044    private static final Logger LOGGER = getLogger(LogbackAuditor.class);
045
046    @Inject
047    private EventBus eventBus;
048
049    /**
050     * Register with the EventBus to receive events.
051     */
052    @PostConstruct
053    public void register() {
054        LOGGER.debug("Initializing: {}", this.getClass().getCanonicalName());
055        eventBus.register(this);
056        LOGGER.warn("The fcrepo-audit extension is now deprecated and will no longer be supported in Fedora 5.x.");
057    }
058
059    @Override
060    @Subscribe
061    public void recordEvent(final FedoraEvent e) {
062        LOGGER.info("{} {}", e.getUserID(), e.getPath());
063    }
064}
